11/// Main entry point for formatting tabular data
12pub fn format_table(
13    data: Vec<Vec<String>>,
14    headers: Option<Vec<String>>,
15    options: &FormatOptions,
16) -> Result<String, Box<dyn Error>> {
17    if data.is_empty() {
18        return Ok("No data to display".to_string());
19    }
20
21    let mut output = String::new();
22    
23    // Add title if provided
24    if let Some(ref title) = options.title {
25        output.push_str(&format!("{}\n\n", title));
26    }
27    
28    // Calculate dimensions
29    let rows = data.len();
30    let cols = data.get(0).map(|row| row.len()).unwrap_or(0);
31    
32    // Add dimensions info if enabled
33    if !options.no_dimensions {
34        // Align with first column (same as row numbers)
35        if !options.no_row_numbering {
36            output.push_str("        ");
37        }
38        
39        let dims_line = if options.use_color {
40            format_dimensions_colored(rows, cols, &options.colors)
41        } else {
42            format!("tv dim: {} x {}\n", rows, cols)
43        };
44        output.push_str(&dims_line);
45    }
46    
47    // Limit rows if max_rows is set
48    let display_rows = if let Some(max) = options.max_rows {
49        rows.min(max)
50    } else {
51        rows
52    };
53    
54    // Transpose data for column-wise operations
55    let mut columns: Vec<Vec<&str>> = vec![vec![]; cols];
56    for row in data.iter().take(display_rows) {
57        for (col_idx, cell) in row.iter().enumerate() {
58            if col_idx < cols {
59                columns[col_idx].push(cell.as_str());
60            }
61        }
62    }
63    
64    // Format columns
65    let formatted_columns: Vec<Vec<String>> = columns
66        .iter()
67        .map(|col| {
68            format_strings(
69                col,
70                options.min_col_width,
71                options.max_col_width,
72                options.significant_figures as i64,
73                options.preserve_scientific,
74                options.max_decimal_width,
75            )
76        })
77        .collect();
78    
79    // Calculate column widths
80    let column_widths: Vec<usize> = formatted_columns
81        .iter()
82        .enumerate()
83        .map(|(idx, col)| {
84            let header_width = if let Some(ref h) = headers {
85                h.get(idx).map(|s| UnicodeWidthStr::width(s.as_str())).unwrap_or(0)
86            } else {
87                0
88            };
89            let content_width = calculate_column_width(col, options.min_col_width, options.max_col_width);
90            header_width.max(content_width)
91        })
92        .collect();
93    
94    // Add row numbers column if enabled
95    let _total_width = if !options.no_row_numbering {
96        let row_num_width = format!("{}", display_rows).len() + 1;
97        column_widths.iter().sum::<usize>() + column_widths.len() * 3 + row_num_width
98    } else {
99        column_widths.iter().sum::<usize>() + column_widths.len() * 3
100    };
101    
102    // Format header using the same logic as data rows
103    if let Some(ref headers) = headers {
104        // Format headers using the same logic as data
105        let header_cols: Vec<Vec<&str>> = headers.iter().map(|h| vec![h.as_str()]).collect();
106        let formatted_headers: Vec<Vec<String>> = header_cols
107            .iter()
108            .map(|col| {
109                format_strings(
110                    col,
111                    options.min_col_width,
112                    options.max_col_width,
113                    options.significant_figures as i64,
114                    options.preserve_scientific,
115                    options.max_decimal_width,
116                )
117            })
118            .collect();
119        
120        let header_row = format_data_row_from_columns(
121            &formatted_headers,
122            0, // First (and only) row
123            0, // No row number for header
124            &column_widths,
125            options
126        );
127        output.push_str(&header_row);
128        output.push('\n');
129    }
130    
131    // Format data rows using the pre-formatted columns
132    for row_idx in 0..display_rows {
133        let formatted_row = format_data_row_from_columns(
134            &formatted_columns, 
135            row_idx, 
136            row_idx + 1, 
137            &column_widths, 
138            options
139        );
140        output.push_str(&formatted_row);
141        output.push('\n');
142    }
143    
144    // Add footer if provided
145    if let Some(ref footer) = options.footer {
146        output.push_str(&format!("\n{}", footer));
147    }
148    
149            // Add "more rows" indicator if truncated
150        if display_rows < rows {
151            let remaining = rows - display_rows;
152            output.push_str(&format!("â€¦ with {} more rows", remaining));
153        }
154    
155    Ok(output)
156}

158/// Format CSV file from path
159pub fn format_csv_file(
160    file_path: &str,
161    options: &FormatOptions,
162) -> Result<String, Box<dyn Error>> {
163    let file = File::open(file_path)?;
164    
165    // Handle empty delimiter by using comma as default
166    let delimiter = if options.delimiter.is_empty() {
167        b','
168    } else {
169        options.delimiter.as_bytes()[0]
170    };
171    
172    let mut reader = ReaderBuilder::new()
173        .delimiter(delimiter)
174        .from_reader(BufReader::new(file));
175    
176    // Read headers
177    let headers = reader.headers()?.iter().map(|s| s.to_string()).collect();
178    
179    // Read data
180    let mut data = Vec::new();
181    for result in reader.records() {
182        let record = result?;
183        let row: Vec<String> = record.iter().map(|s| s.to_string()).collect();
184        data.push(row);
185    }
186    
187    format_table(data, Some(headers), options)
188}

190/// Format Parquet file from path
191pub fn format_parquet_file(
192    file_path: &str,
193    options: &FormatOptions,
194) -> Result<String, Box<dyn Error>> {
195    use parquet::file::reader::{FileReader, SerializedFileReader};
196    
197    let file = File::open(file_path)?;
198    let reader = SerializedFileReader::new(file)?;
199    let iter = reader.get_row_iter(None)?;
200
201    let mut data = Vec::new();
202    let mut headers = Vec::new();
203
204    // Extract column names from schema
205    let schema = reader.metadata().file_metadata().schema_descr();
206    let mut column_indices_to_include = Vec::new();
207
208    for i in 0..schema.num_columns() {
209        let column = schema.column(i);
210        let column_name = column.name().to_lowercase();
211
212        // Skip columns that are likely pandas index columns
213        if column_name == "id" || column_name == "index" || column_name == "__index_level_0__" {
214            continue;
215        }
216
217        headers.push(column.name().to_string());
218        column_indices_to_include.push(i);
219    }
220
221    // Process all data rows
222    for row_result in iter {
223        let row = row_result?;
224        let mut record_fields = Vec::new();
225
226        for &col_index in &column_indices_to_include {
227            if let Some(field) = row.get_column_iter().nth(col_index) {
228                let value_str = format!("{}", field.1);
229                // Remove quotes from string values to match CSV behavior
230                let clean_value = if value_str.starts_with('"')
231                    && value_str.ends_with('"')
232                    && value_str.len() > 1
233                {
234                    value_str[1..value_str.len() - 1].to_string()
235                } else {
236                    value_str
237                };
238                record_fields.push(clean_value);
239            } else {
240                record_fields.push(String::new());
241            }
242        }
243        data.push(record_fields);
244    }
245
246    format_table(data, Some(headers), options)
247}

249/// Format Arrow file from path
250pub fn format_arrow_file(
251    file_path: &str,
252    options: &FormatOptions,
253) -> Result<String, Box<dyn Error>> {
254    use arrow::array::{Array, BooleanArray, Float64Array, Int64Array, StringArray};
255    use arrow::datatypes::DataType;
256    use arrow::error::ArrowError;
257    use arrow::ipc::reader::FileReader as ArrowFileReader;
258    
259    let file = File::open(file_path)?;
260
261    // Try to read as uncompressed first
262    let reader = match ArrowFileReader::try_new(file, None) {
263        Ok(reader) => reader,
264        Err(ArrowError::InvalidArgumentError(msg)) if msg.contains("lz4") => {
265            return Err("Arrow files with LZ4 compression are not supported".into());
266        }
267        Err(e) => return Err(e.into()),
268    };
269
270    let schema = reader.schema();
271
272    let mut headers = Vec::new();
273    let mut data = Vec::new();
274
275    // Extract column names from schema
276    for field in schema.fields() {
277        headers.push(field.name().to_string());
278    }
279
280    // Read all batches and convert to StringRecords
281    for batch_result in reader {
282        let batch = batch_result?;
283        let num_rows = batch.num_rows();
284        let num_cols = batch.num_columns();
285
286        for row_idx in 0..num_rows {
287            let mut row_data = Vec::new();
288            for col_idx in 0..num_cols {
289                let array = batch.column(col_idx);
290                let value = match array.data_type() {
291                    DataType::Utf8 => {
292                        let string_array = array.as_any().downcast_ref::<StringArray>().unwrap();
293                        if array.is_null(row_idx) {
294                            "NA".to_string()
295                        } else {
296                            string_array.value(row_idx).to_string()
297                        }
298                    }
299                    DataType::Int64 => {
300                        let int_array = array.as_any().downcast_ref::<Int64Array>().unwrap();
301                        if array.is_null(row_idx) {
302                            "NA".to_string()
303                        } else {
304                            int_array.value(row_idx).to_string()
305                        }
306                    }
307                    DataType::Float64 => {
308                        let float_array = array.as_any().downcast_ref::<Float64Array>().unwrap();
309                        if array.is_null(row_idx) {
310                            "NA".to_string()
311                        } else {
312                            float_array.value(row_idx).to_string()
313                        }
314                    }
315                    DataType::Boolean => {
316                        let bool_array = array.as_any().downcast_ref::<BooleanArray>().unwrap();
317                        if array.is_null(row_idx) {
318                            "NA".to_string()
319                        } else {
320                            bool_array.value(row_idx).to_string()
321                        }
322                    }
323                    _ => {
324                        // For other types, convert to string representation
325                        "NA".to_string()
326                    }
327                };
328                row_data.push(value);
329            }
330            data.push(row_data);
331        }
332    }
333
334    format_table(data, Some(headers), options)
335}

337fn format_dimensions_colored(rows: usize, cols: usize, colors: &ColorScheme) -> String {

340        "tv dim: {} x {}\n",

346fn format_header_row(headers: &[String], widths: &[usize], options: &FormatOptions) -> String {
347    let mut row = String::new();
348    
349    if !options.no_row_numbering {
350        row.push_str("     ");
351    }
352    
353    for (idx, (header, width)) in headers.iter().zip(widths.iter()).enumerate() {
354        let padded = format!("{:^width$}", header, width = width);
355        if options.use_color {
356            let [r, g, b] = options.colors.header_color;
357            row.push_str(&padded.truecolor(r, g, b).to_string());
358        } else {
359            row.push_str(&padded);
360        }
361    }
362    
363    row
364}

366fn format_data_row_from_columns(
367    formatted_columns: &[Vec<String>],
368    row_idx: usize,
369    row_num: usize,
370    widths: &[usize],
371    options: &FormatOptions,
372) -> String {
373    let mut output = String::new();
374    
375    if !options.no_row_numbering && row_num > 0 {
376        let row_num_str = format!("{: >6}  ", row_num);
377        if options.use_color {
378            let [r, g, b] = options.colors.meta_color;
379            output.push_str(&row_num_str.truecolor(r, g, b).to_string());
380        } else {
381            output.push_str(&row_num_str);
382        }
383    } else if !options.no_row_numbering {
384        // For headers, add the same spacing as row numbers
385        output.push_str("        ");
386    }
387    
388    for (col_idx, formatted_col) in formatted_columns.iter().enumerate() {
389        // Handle uneven rows by providing a default value if the cell doesn't exist
390        let cell = if row_idx < formatted_col.len() {
391            &formatted_col[row_idx]
392        } else {
393            "NA"  // Default value for missing cells
394        };
395        
396        let width = widths.get(col_idx).unwrap_or(&0);
397        
398        let padded = format!("{:<width$}", cell, width = width);
399        
400        if options.use_color {
401            let colored = if row_num == 0 {
402                // Header row - use header color
403                let [r, g, b] = options.colors.header_color;
404                padded.truecolor(r, g, b).to_string()
405            } else if is_na(cell) {
406                let [r, g, b] = options.colors.na_color;
407                padded.truecolor(r, g, b).to_string()
408            } else if is_negative_number(cell) {
409                let [r, g, b] = options.colors.neg_num_color;
410                padded.truecolor(r, g, b).to_string()
411            } else {
412                let [r, g, b] = options.colors.std_color;
413                padded.truecolor(r, g, b).to_string()
414            };
415            output.push_str(&colored);
416        } else {
417            output.push_str(&padded);
418        }
419    }
420    
421    output
422}
