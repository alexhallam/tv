14/// Python class for formatting options
15#[pyclass]
16#[derive(Clone)]
17pub struct PyFormatOptions {
18    inner: FormatOptions,
19}
20
21#[pymethods]
22impl PyFormatOptions {
23    #[new]
24    #[pyo3(signature = (
25        max_rows=None,
26        max_col_width=20,
27        min_col_width=2,
28        use_color=true,
29        color_theme="nord",
30        delimiter=",",
31        significant_figures=3,
32        preserve_scientific=false,
33        max_decimal_width=13,
34        no_dimensions=false,
35        no_row_numbering=false,
36        title=None,
37        footer=None
38    ))]
39    pub fn new(
40        max_rows: Option<usize>,
41        max_col_width: usize,
42        min_col_width: usize,
43        use_color: bool,
44        color_theme: &str,
45        delimiter: &str,
46        significant_figures: usize,
47        preserve_scientific: bool,
48        max_decimal_width: usize,
49        no_dimensions: bool,
50        no_row_numbering: bool,
51
52        title: Option<String>,
53        footer: Option<String>,
54    ) -> PyResult<Self> {
55        let colors = match color_theme {
56            "nord" => ColorScheme::nord(),
57            "one_dark" => ColorScheme::one_dark(),
58            "gruvbox" => ColorScheme::gruvbox(),
59            "dracula" => ColorScheme::dracula(),
60            "solarized_light" => ColorScheme::solarized_light(),
61            _ => ColorScheme::nord(),
62        };
63
64        Ok(PyFormatOptions {
65            inner: FormatOptions {
66                max_rows,
67                max_col_width,
68                min_col_width,
69                use_color,
70                colors,
71                delimiter: delimiter.to_string(),
72                significant_figures,
73                preserve_scientific,
74                max_decimal_width,
75                no_dimensions,
76                no_row_numbering,
77
78                title,
79                footer,
80            },
81        })
82    }
83
84    /// Set color theme
85    pub fn set_color_theme(&mut self, theme: &str) -> PyResult<()> {
86        self.inner.colors = match theme {
87            "nord" => ColorScheme::nord(),
88            "one_dark" => ColorScheme::one_dark(),
89            "gruvbox" => ColorScheme::gruvbox(),
90            "dracula" => ColorScheme::dracula(),
91            "solarized_light" => ColorScheme::solarized_light(),
92            _ => return Err(PyErr::new::<pyo3::exceptions::PyValueError, _>(
93                format!("Unknown color theme: {}", theme)
94            )),
95        };
96        Ok(())
97    }
98}

100/// Format tabular data from Python lists
101#[pyfunction]
102#[pyo3(signature = (data, headers=None, options=None))]
103pub fn format_data(
104    data: Vec<Vec<String>>,
105    headers: Option<Vec<String>>,
106    options: Option<&PyFormatOptions>,
107) -> PyResult<String> {
108    let format_options = if let Some(opts) = options {
109        &opts.inner
110    } else {
111        &FormatOptions::default()
112    };
113
114    format_table(data, headers, format_options)
115        .map_err(|e| PyErr::new::<pyo3::exceptions::PyRuntimeError, _>(e.to_string()))
116}

171/// Format a dictionary of lists
172#[pyfunction]
173pub fn format_dict_of_lists(data_dict: &Bound<'_, PyAny>, options: Option<&PyFormatOptions>) -> PyResult<String> {
174    let dict: HashMap<String, Vec<String>> = data_dict.extract()?;
175    
176    if dict.is_empty() {
177        return Err(PyErr::new::<pyo3::exceptions::PyValueError, _>(
178            "Dictionary is empty"
179        ));
180    }
181    
182    // Get headers (keys)
183    let headers: Vec<String> = dict.keys().cloned().collect();
184    
185    // Get maximum length
186    let max_len = dict.values().map(|v| v.len()).max().unwrap_or(0);
187    
188    // Convert to list of lists
189    let mut data = Vec::new();
190    for i in 0..max_len {
191        let mut row = Vec::new();
192        for header in &headers {
193            let na_value = "NA".to_string();
194            let value = dict.get(header)
195                .and_then(|v| v.get(i))
196                .unwrap_or(&na_value);
197            row.push(value.clone());
198        }
199        data.push(row);
200    }
201    
202    format_data(data, Some(headers), options)
203}

205/// Format a list of dictionaries
206#[pyfunction]
207pub fn format_list_of_dicts(data_list: &Bound<'_, PyAny>, options: Option<&PyFormatOptions>) -> PyResult<String> {
208    let list: Vec<HashMap<String, String>> = data_list.extract()?;
209    
210    if list.is_empty() {
211        return Err(PyErr::new::<pyo3::exceptions::PyValueError, _>(
212            "List is empty"
213        ));
214    }
215    
216    // Get all unique keys
217    let mut all_keys = std::collections::HashSet::new();
218    for dict in &list {
219        all_keys.extend(dict.keys().cloned());
220    }
221    let headers: Vec<String> = all_keys.into_iter().collect();
222    
223    // Convert to list of lists
224    let mut data = Vec::new();
225    for dict in &list {
226        let mut row = Vec::new();
227        for header in &headers {
228            let na_value = "NA".to_string();
229            let value = dict.get(header).unwrap_or(&na_value);
230            row.push(value.clone());
231        }
232        data.push(row);
233    }
234    
235    format_data(data, Some(headers), options)
236}
